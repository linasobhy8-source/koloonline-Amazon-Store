import express from "express"; import mongoose from "mongoose"; import cors from "cors"; import dotenv from "dotenv"; dotenv.config(); const app = express(); app.use(cors()); app.use(express.json()); /* ================= MONGO CONNECT ================= */ mongoose.connect(process.env.MONGODB_URI) .then(()=>console.log("MongoDB Connected")) .catch(err=>console.log(err)); /* ================= SCHEMA ================= */ const AnalyticsSchema = new mongoose.Schema({ asin:String, click:Number, cart:Number, whatsapp:Number, country:Object, createdAt:{type:Date,default:Date.now} }); const Analytics = mongoose.model("Analytics",AnalyticsSchema); /* ================= TRACK EVENT API ================= */ app.post("/track", async (req,res)=>{ try{ const {asin,type,country} = req.body; let doc = await Analytics.findOne({asin}); if(!doc){ doc = new Analytics({ asin, click:0, cart:0, whatsapp:0, country:{} }); } if(type){ doc[type] = (doc[type]||0) + 1; } if(country){ doc.country[country] = (doc.country[country]||0)+1; } await doc.save(); res.json({success:true}); }catch(err){ res.json({success:false}); } }); /* ================= GET DASHBOARD DATA ================= */ app.get("/analytics", async (req,res)=>{ const data = await Analytics.find(); res.json(data); }); /* ================= SERVER ================= */ app.listen(5000, ()=> console.log("Server running on port 5000"));
